    public float kembalian(float bayar, float total) {
        float kembali = bayar - total;
        lblKembali.setText(Float.toString(kembali));
        DecimalFormat angka = new DecimalFormat("###,###");
        lblKembali.setText("Rp "+angka.format(kembali));
        return kembali;
    }
